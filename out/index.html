Model Builder

A simple interface to build neural networks in TensorFlow.js. Here's a demo implementation:

const { TFJSModel } = await import("https://episphere.github.io/tfjs/modelBuilder.js")
const model = new TFJSModel({
    'architecture': [5,3,3],
    'numEpochs': 50,
    'lossFunc': "categoricalCrossentropy",
    'metrics': ["accuracy"]
})

await model.loadData({
    "datasetURL": "https://episphere.github.io/ai/data/irisOHE.json",
    "labelFieldName": "species",
    "trainTestRatio": 0.8
})

await model.build()

await model.train()

const testAccuracy = await model.test()
console.log(`Final model test accuracy: ${testAccuracy}`)
